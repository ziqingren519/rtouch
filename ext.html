tmove函数扩展


  function tmove(o){   
    if (o.dir === "x"||"y") { alert("错误，dir为x或y"); return;};
    var span = o.span,len = span.length > 1 ? span.length : parseInt(span.innerHTML),dir = o.dir || "x",con = o.con,set = -o.set || (dir === "x" ? -con.offsetWidth : -con.offsetHeight )/len,index = 0,distance,base = -set/4,autorun = o.autorun,movestates,setI,obj;
    function go(i){
      index = i;
      var tran = dir === "x" ? i*set +"px,0px" : "0px,"+i*set +"px" ; 
      con.style.Transform = "translate("+ i*set +"px)";
      con.style.WebkitTransform = "translate3d("+ i*set +"px,0px,0px)";
      o.fn ? o.fn(i) : 1;
    };
    if (span.length > 1) {
      for (var i = 0; i < span.length; i++) {
        span[i].onclick = function(i){
          return function(){
            run(i);
          };
        }(i);
      };
    }else{
      span.innerHTML = "1"+"/"+len;
    };

    function run(i){
      i = i < 0 ? 0 : i > (len-1) ? (len-1) : i;
      if (span.length > 1) {
        for (var a = 0; a < span.length; a++) {
          span[a].className = "";
        };
        span[i].className = "hover";
      }else {
        span.innerHTML = ""+(i+1)+""+"/"+len;
      }
      go(i);
    };

    function autorunfun(){
      setI = setInterval(function(){
        index = index+1 > (len-1) ? 0 : index+1;
        run(index);
      },autorun);
    };
    function move(e){
      if ((e.type === "holdleft" && index >= len-1) || (e.type === "holdright" && index <= 0)) { return;}
      movestates = 1;
      distance = e.x-e.cachedX ;
      con.style.Transform = "translate("+ (index*set+distance) +"px)";
      con.style.WebkitTransform = "translate3d("+ (index*set+distance) +"px,0px,0px)";
    }
    
    obj = new Rtouch(con);
    obj.swipestart = function(e){
      con.style.WebkitTransitionDuration = "0s";
      con.style.TransitionDuration= "0s";
      autorun ? clearInterval(setI) : 1 ;
    };
    
    dir === "x" ? obj.holdx = move : obj.holdy = move;
    obj.swipeend=function(e){
      con.style.WebkitTransitionDuration = "0.3s";
      con.style.TransitionDuration= "0.3s";
      if (!movestates) {return};
      -distance > base ? run(index+1) : distance > base ? run(index-1) : run(index);
      movestates = 0;
      autorun ? autorunfun() : 1 ;
    };
    autorun ? autorunfun() : 1;
  };

tmove配置

tmove(tmove配置数组，Rtouch配置数组 可选)

  • 配置项介绍
  • span[必选] 传过来索引的数组，例：document.getElementById('spanbox').getElementsByTagName('span')
  • silde[必选] 内容的最外层对象
  • sildebox[必选] 承载内容的对象 主要靠它滚动
  • dir[可选 "x"或"y" 默认为"x"] "x"为横向滚动 "y"为竖向滚动
  • spanclass[可选 默认为"hover"] 当前索引所添加的class名称
  • fn[可选] 回调函数 每次滚动后执行后的函数
  • autorun[可选 默认不自动运行] 是否自动运行 如过自动运行则传来毫秒运行的时间 例：3000 (每隔3秒自动运行)
  • set[可选 默认"总宽度/索引数量"] 每次滚动的距离
  • base[可选 默认"超过单次滚动的1/4距离"] 最小触摸滚动的距离
  • time[可选 默认"300ms"] 滚动所花费的时间
