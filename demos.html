swipeleft - swiperight - swipetop - swipedown 例子

swipe

  var box = document.getElementById('box1'),
      ball = document.getElementById('ball1'),
      rtouch = new Rtouch(box,{
        PREDEF : [1,1]
      });
      rtouch.swipeleft = function(e){
        ball.style.webkitTransform = 'translate3d(-300px,0,0)'; 
      }
      rtouch.swiperight = function(e){
        ball.style.webkitTransform = 'translate3d(300px,0,0)'; 
      }
      rtouch.swipetop = function(e){
        ball.style.webkitTransform = 'translate3d(0,-100px,0)'; 
      }
      rtouch.swipedown = function(e){
        ball.style.webkitTransform = 'translate3d(0,100px,0)'; 
      }

holdx - holdy 例子

holdx(y)

  var ball = document.getElementById('ball2'),
      rtouch = new Rtouch(ball,{
        PREDEF : [1,1]
      });
      rtouch.holdx = function(e){
        ball.style.webkitTransform = 'translate3d('+(e.x-e.cX)+'px,0,0)'; 
      }
      rtouch.holdy = function(e){
        ball.style.webkitTransform = 'translate3d(0,'+(e.y-e.cY)+'px,0)'; 
      }
